**Goal:** Reemplazar el REPL de readline con un servidor Express que sirve un chat web con diseño fiel a ui.md, historial persistente y streaming SSE.

**Architecture:** Express en `server.ts` sirve archivos estáticos de `public/` y expone 4 rutas API. La lógica del agente SDK se extrae a `agent.ts` como generador async. El frontend en HTML/CSS/JS vanilla consume el stream SSE via `fetch` + `ReadableStream`.

**Tech Stack:** TypeScript, Express, `@anthropic-ai/claude-agent-sdk`, HTML/CSS/JS vanilla, SSE

> **Nota sobre tests:** El proyecto no tiene suite de tests configurada. Las verificaciones se hacen con `curl` para la API y revisión manual en el browser para la UI.

```typescript
import { query } from "@anthropic-ai/claude-agent-sdk";
import * as dotenv from "dotenv";

dotenv.config();

const SYSTEM_PROMPT =
  "Eres un asistente especializado en bases de datos Supabase. " +
  "Consulta EXCLUSIVAMENTE el schema 'public'. " +
  "No accedas ni menciones ningún otro schema (auth, storage, extensions, etc.). " +
  "Si una consulta requiriese datos fuera del schema public, indícalo al usuario en lugar de acceder a otros schemas.";

const mcpOptions = {
  mcpServers: {
    supabase: {
      type: "http" as const,
      url: "https://mcp.supabase.com/mcp?project_ref=nfjjlfovpznoipgkugdf&read_only=true",
      headers: {
        Authorization: `Bearer ${process.env.SUPABASE_ACCESS_TOKEN}`,
      },
    },
  },
  allowedTools: ["mcp__supabase__*"],
  permissionMode: "bypassPermissions" as const,
  system: SYSTEM_PROMPT,
};

export type Message = { role: "user" | "assistant"; content: string };

export async function* runAgent(
  prompt: string,
  history: Message[]
): AsyncGenerator<string> {
  let fullPrompt = prompt;
  if (history.length > 0) {
    const historyText = history
      .map((m) => `${m.role === "user" ? "Usuario" : "Asistente"}: ${m.content}`)
      .join("\n");
    fullPrompt = `[Conversación previa]\n${historyText}\n\n[Mensaje actual]\nUsuario: ${prompt}`;
  }

  for await (const message of query({ prompt: fullPrompt, options: mcpOptions })) {
    if (message.type === "assistant") {
      const content = (message as any).message?.content ?? [];
      const text = content
        .filter((b: any) => b.type === "text")
        .map((b: any) => b.text)
        .join("");
      if (text) yield text;
    }
  }
}
```

```typescript
import express from "express";
import * as path from "path";
import * as fs from "fs";
import * as dotenv from "dotenv";
import { runAgent, Message } from "./agent";

dotenv.config();

if (!process.env.SUPABASE_ACCESS_TOKEN || !process.env.ANTHROPIC_API_KEY) {
  console.error("ERROR: Faltan variables de entorno: ANTHROPIC_API_KEY y/o SUPABASE_ACCESS_TOKEN.");
  process.exit(1);
}

const app = express();
const PORT = Number(process.env.PORT) || 3000;
const HISTORY_FILE = path.join(process.cwd(), "chat-history.json");

app.use(express.json());
app.use(express.static(path.join(process.cwd(), "public")));

function readHistory(): Message[] {
  try {
    return JSON.parse(fs.readFileSync(HISTORY_FILE, "utf-8"));
  } catch {
    return [];
  }
}

function writeHistory(history: Message[]): void {
  fs.writeFileSync(HISTORY_FILE, JSON.stringify(history, null, 2));
}

app.get("/api/history", (_req, res) => {
  res.json(readHistory());
});

app.delete("/api/history", (_req, res) => {
  writeHistory([]);
  res.json({ ok: true });
});

app.post("/api/chat", async (req, res) => {
  const { message } = req.body as { message: string };
  if (!message?.trim()) {
    res.status(400).json({ error: "message required" });
    return;
  }

  const history = readHistory();

  res.setHeader("Content-Type", "text/event-stream");
  res.setHeader("Cache-Control", "no-cache");
  res.setHeader("Connection", "keep-alive");
  res.flushHeaders();

  let fullResponse = "";

  try {
    for await (const chunk of runAgent(message, history)) {
      fullResponse += chunk;
      res.write(`data: ${JSON.stringify(chunk)}\n\n`);
    }
  } catch (err) {
    console.error("[agent error]", err);
    res.write(`data: "[ERROR: el agente no pudo responder]"\n\n`);
    fullResponse = "[ERROR: el agente no pudo responder]";
  }

  writeHistory([
    ...history,
    { role: "user", content: message },
    { role: "assistant", content: fullResponse },
  ]);

  res.write(`data: [DONE]\n\n`);
  res.end();
});

app.listen(PORT, () => {
  console.log(`=========================================`);
  console.log(`Agente Supabase — http://localhost:${PORT}`);
  console.log(`=========================================`);
});
```

```javascript
const messagesEl = document.getElementById('messages');
const inputEl = document.getElementById('messageInput');
const sendBtn = document.getElementById('sendBtn');
const newChatBtn = document.getElementById('newChatBtn');

// ── Helpers de render ────────────────────────────────────────────

function renderUserBubble(text) {
  const row = document.createElement('div');
  row.className = 'msg-user';
  const bubble = document.createElement('div');
  bubble.className = 'bubble-user';
  bubble.textContent = text;
  row.appendChild(bubble);
  messagesEl.appendChild(row);
  scrollToBottom();
  return bubble;
}

function renderAssistantBubble(text = '') {
  const row = document.createElement('div');
  row.className = 'msg-assistant';
  const bubble = document.createElement('div');
  bubble.className = 'bubble-assistant';
  bubble.textContent = text;
  row.appendChild(bubble);
  messagesEl.appendChild(row);
  scrollToBottom();
  return bubble;
}

function renderTyping() {
  const row = document.createElement('div');
  row.className = 'msg-typing';
  row.id = 'typingIndicator';
  row.innerHTML = `
    <div class="typing-bubble">
      <div class="typing-dot"></div>
      <div class="typing-dot"></div>
      <div class="typing-dot"></div>
    </div>`;
  messagesEl.appendChild(row);
  scrollToBottom();
}

function removeTyping() {
  const el = document.getElementById('typingIndicator');
  if (el) el.remove();
}

function scrollToBottom() {
  messagesEl.scrollTop = messagesEl.scrollHeight;
}

function setInputEnabled(enabled) {
  inputEl.disabled = !enabled;
  sendBtn.disabled = !enabled;
}

// ── Carga del historial al iniciar ───────────────────────────────

async function loadHistory() {
  try {
    const res = await fetch('/api/history');
    const history = await res.json();
    for (const msg of history) {
      if (msg.role === 'user') renderUserBubble(msg.content);
      else renderAssistantBubble(msg.content);
    }
  } catch (err) {
    console.error('Error cargando historial:', err);
  }
}

// ── Envío de mensaje con streaming SSE ───────────────────────────

async function sendMessage() {
  const text = inputEl.value.trim();
  if (!text) return;

  inputEl.value = '';
  setInputEnabled(false);

  renderUserBubble(text);
  renderTyping();

  let assistantBubble = null;
  let buffer = '';

  try {
    const response = await fetch('/api/chat', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ message: text }),
    });

    if (!response.ok) throw new Error(`HTTP ${response.status}`);

    const reader = response.body.getReader();
    const decoder = new TextDecoder();

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;

      buffer += decoder.decode(value, { stream: true });

      // Procesar líneas completas del formato SSE ("data: ...\n\n")
      const lines = buffer.split('\n');
      buffer = lines.pop() ?? ''; // guarda línea incompleta

      for (const line of lines) {
        if (!line.startsWith('data: ')) continue;
        const payload = line.slice(6);

        if (payload === '[DONE]') {
          removeTyping();
          setInputEnabled(true);
          inputEl.focus();
          return;
        }

        try {
          const chunk = JSON.parse(payload);
          removeTyping();
          if (!assistantBubble) {
            assistantBubble = renderAssistantBubble('');
          }
          assistantBubble.textContent += chunk;
          scrollToBottom();
        } catch {
          // línea SSE no es JSON válido, ignorar
        }
      }
    }
  } catch (err) {
    console.error('Error en streaming:', err);
    removeTyping();
    renderAssistantBubble('Error al conectar con el agente. Intentá de nuevo.');
  }

  setInputEnabled(true);
  inputEl.focus();
}

// ── Nueva conversación ────────────────────────────────────────────

newChatBtn.addEventListener('click', async () => {
  try {
    await fetch('/api/history', { method: 'DELETE' });
    messagesEl.innerHTML = '';
    inputEl.focus();
  } catch (err) {
    console.error('Error limpiando historial:', err);
  }
});

// ── Event listeners ───────────────────────────────────────────────

sendBtn.addEventListener('click', sendMessage);

inputEl.addEventListener('keydown', (e) => {
  if (e.key === 'Enter' && !e.shiftKey) {
    e.preventDefault();
    sendMessage();
  }
});

// ── Inicializar ───────────────────────────────────────────────────

loadHistory();
inputEl.focus();
```

```markdown
## Architecture

Two-file TypeScript backend (`server.ts` + `agent.ts`) serving a vanilla HTML/CSS/JS frontend:

1. **`agent.ts`** — exports `runAgent(prompt, history)`, an async generator that calls `query()` from the Claude Agent SDK and yields text chunks. Prepends conversation history to the prompt string for context.

2. **`server.ts`** — Express server (default port 3000) with four routes:
   - `GET /` — serves `public/index.html`
   - `GET /api/history` — returns `chat-history.json` (or `[]`)
   - `POST /api/chat` — SSE stream: calls `runAgent()`, sends each chunk as `data: <json>\n\n`, saves pair to `chat-history.json` on completion
   - `DELETE /api/history` — clears `chat-history.json`

3. **`public/`** — static frontend:
   - `index.html`: chat layout (header, messages area, input)
   - `style.css`: pill-shaped bubbles, user messages right-aligned blue, assistant messages centered gray
   - `app.js`: loads history on start, sends messages via `fetch` + `ReadableStream` (SSE), streams chunks into assistant bubble in real time

4. **`chat-history.json`** — generated at runtime (gitignored). Array of `{role, content}` objects, read fresh on each request.
```
